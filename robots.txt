# =========================================================
# VioniaSoft 网站 robots.txt
# 兼容 Google、Naver 等主流搜索引擎
# 放置位置：网站根目录 (与 index.html 同级)
# =========================================================

# ==========================
# 所有搜索引擎都适用
# ==========================
User-agent: *
# Disallow: /private/    # 如果有不希望抓取的目录，可取消注释并修改
Disallow:                   # 空白表示允许抓取所有内容

# ==========================
# 指定 Naver bot（可选）
# ==========================
User-agent: Yeti
Disallow: /private/         # 如果有私密目录不希望 Naver 抓取
# Disallow:                 # 如果允许抓取所有内容

# ==========================
# Sitemap 提交
# ==========================
Sitemap: https://www.vioniasoft.com/sitemap.xml
# 注意：本地测试可改为本地路径，例如：
# Sitemap: http://localhost:8000/sitemap.xml

# ==========================
# 说明：
# 1. 文件名必须小写：robots.txt
# 2. 必须放在网站根目录，GitHub Pages 根目录亦是如此
# 3. 禁止抓取目录可在 Disallow: 后填写路径，例如：
#    Disallow: /admin/
